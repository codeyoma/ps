    vector<int> v(n);

    for (int i = 0; i < n; ++i) {

    }

    sort(v.begin(), v.end());

    cin >> m;

    for (int i = 0; i < m; ++i) {
        int t;
        cin >> t;

        auto it = lower_bound(v.begin(), v.end(), t);
