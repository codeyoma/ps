    fast_io();

    //   logic
    // 1  2  3  4   5   6
    // 1  3  6 10  15  21
    // 1  4 10 20  35  56
    // 1  5 15 35  70 126
    // 1  6 21 56 126 252

    vector<vector<int>> dp(15, vector<int>(15));

    for (int i = 1; i < 15; ++i) {
        dp[0][i] = i;
    }

    for (int i = 1; i < 15; ++i) {
        for (int j = 1; j < 15; ++j) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
        }
    }

    int t;
    cin >> t;

    while (t--) {
        int k, n;
        cin >> k >> n;
        cout << dp[k][n] << "\n";
    }
}
