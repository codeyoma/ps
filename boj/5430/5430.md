// https://www.acmicpc.net/problem/5430
// https://codeyoma.github.io/Computer-Science/1-Foundations--and--Theory/Algorithms/ps/boj/5430/5430
#include <iostream>
using namespace std;

#ifdef LOCAL
#    define LOG clog
#else
struct nullstream : ostream {
    nullstream()
        : ostream(nullptr) {}
};
nullstream LOG;
#endif

//--------------------------------------------------------------------------------------------------

#define MAX (1234567891)
#define MIN (-1234567891)

#include <deque>
#include <iostream>

int main() {
    //   logic
    int t;
    cin >> t;

    while (t--) {
        int    n;
        string p, array;

        cin >> p >> n >> array;

        // parse array
        // excute p cmd
        // print result

        deque<string> q;

        {
            int l = 0, r = 0;

            while (l < array.size() && r < array.size()) {
                if (n == 0) {
                    break;
                } else if (array[l] == '[') {
                    l = 1;
                    r = l;
                } else if (array[r] == ',' || array[r] == ']') {
                    string temp = array.substr(l, r - l);
                    q.push_back(temp);
                    l = r + 1;
                    r = l;
                } else {
                    r++;
                }
            }
        }

        bool is_front = true;
        bool is_error = false;
        {
            for (const char& c: p) {
                if (c == 'R') {
                    is_front = !is_front;
                } else {
                    if (q.empty()) {
                        is_error = true;
                    } else if (is_front) {
                        q.pop_front();
                    } else {
                        q.pop_back();
                    }
                }
            }
        }

        {
            if (is_error) {
                cout << "error\n";
            } else {
                cout << "[";
                if (is_front) {
                    for (deque<string>::iterator it = q.begin(); it != q.end(); ++it) {
                        cout << *it;
                        if (next(it) != q.end()) {
                            cout << ',';
                        }
                    }
                } else {
                    for (deque<string>::reverse_iterator it = q.rbegin(); it != q.rend(); ++it) {
                        cout << *it;
                        if (next(it) != q.rend()) {
                            cout << ',';
                        }
                    }
                }
                cout << "]\n";
            }
        }
    }
}
