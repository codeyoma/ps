using namespace std;
#define FAST_IO                  \
    ios::sync_with_stdio(false); \
    cin.tie(nullptr);

#define END   \
    return 0;

#include <vector>

template<typename T>
using VV = vector<vector<T>>;

template<typename T>
using V = vector<T>;

// typedef long long ll;
using LL  = long long;
using ULL = unsigned long long;

constexpr int _MAX  = 1'234'567'891; // prime
constexpr int _MIN  = -_MAX;
constexpr LL  __MAX = 1'111'111'111'111'111'111LL; // prime
constexpr LL  __MIN = -__MAX;

//--------------------------------------------------------------------------------------------------

int main() {
    FAST_IO;

    //   logic
    int     n = 10'001;
    V<bool> t(n + 1);

    auto d = [&](int i) {
        if (i > n) {
            return 0;
        }

        int r = i;

        while (i > 0) {
            r += i % 10;
            i /= 10;
        }

        return r;
    };

    for (int i = 1; i <= n; ++i) {
        t[d(i)] = true;

        if (!t[i]) {
            cout << i << "\n";
        }
    }

    END;
}
