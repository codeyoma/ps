using namespace std;
#define FAST_IO                  \
    ios::sync_with_stdio(false); \
    cin.tie(nullptr);

// typedef long long ll;
using ll = long long;

constexpr int _MAX  = 1'234'567'891; // prime
constexpr int _MIN  = -_MAX;
constexpr ll  __MAX = 1'111'111'111'111'111'111LL; // prime
constexpr ll  __MIN = -__MAX;

//--------------------------------------------------------------------------------------------------

#include <string>
#include <vector>

int main() {
    FAST_IO;

    //   logic
    int n;
    cin >> n;

    bool inc = false;
    bool dec = false;

    string before = "", current;

    while (n--) {
        cin >> current;
        if (before != "") {
            if (before < current) {
                inc = true;
            } else {
                dec = true;
            }
        }

        before = current;
    }

    if (inc && dec) {
        cout << "NEITHER";
    } else if (inc) {
        cout << "INCREASING";
    } else {
        cout << "DECREASING";
    }

    return 0;
}
