#include <stack>

int main() {
    //   logic
    // 새로 들어온 것보다 큰게 나올떄 까지 pop, and then push
    // stack.size() = max view size

    int n;
    cin >> n;

    stack<int> s;

    for (int i = 0; i < n; ++i) {
        int temp;
        cin >> temp;

        while (!s.empty() && s.top() <= temp) {
            s.pop();
        }
        s.push(temp);
    }

    cout << s.size();
}
