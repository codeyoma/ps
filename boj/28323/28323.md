#define MAX (1234567891)
#define MIN (-1234567891)

#include <iostream>
#include <vector>

int main() {
    int n;
    cin >> n;
    vector<int> v(n);
    for (int i = 0; i < n; ++i) {
        cin >> v[i];
    }

    int  odd_start_cnt  = 0;
    int  even_start_cnt = 0;
    bool is_odd         = true;
    bool is_even        = true;

    for (const auto& i: v) {
        if (i & 1) {
            if (is_odd) {
                odd_start_cnt++;
                is_odd = !is_odd;
            }

            if (!is_even) {
                even_start_cnt++;
                is_even = !is_even;
            }

        } else {
            if (!is_odd) {
                odd_start_cnt++;
                is_odd = !is_odd;
            }

            if (is_even) {
                even_start_cnt++;
                is_even = !is_even;
            }
        }
    }

    cout << max(odd_start_cnt, even_start_cnt);
}
