1. 알고리즘 선택
	- 맨 오른쪽 찍고 넘어오기
	- 오래 걸리는 값 더하기

2. 자료구조 선택
3. 수도 코드 작성
4. 정합판단
  - `1 ~ 3`과정으로 도출된 로직이 문제를 해결하는가
    - 그렇다 -> **구현**
    - 잘 모르겠다 -> **구현**
    - 아니다 -> 1번부터 다시 점검

    long long n, answer = 0, x_pos = 0;
    cin >> n;

    vector<long long> x(n, 0), t(n, 0);

    for (int i = 0; i < n; ++i) {
        cin >> x[i];
    }
    for (int i = 0; i < n; ++i) {
        cin >> t[i];
    }

    for (int i = n - 1; i >= 0; --i) {
        long long dist = abs(x[i] - x_pos);
        if (t[i] > answer) {
            dist = max(dist, t[i] - answer);
        }
        answer += dist;
        x_pos = x[i];
    }

    answer += x_pos;

    cout << answer;
}
