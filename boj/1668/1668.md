using namespace std;
#define FAST_IO                  \
    ios::sync_with_stdio(false); \
    cin.tie(nullptr);

// typedef long long ll;
using ll = long long;

constexpr int _MAX  = 1'234'567'891; // prime
constexpr int _MIN  = -_MAX;
constexpr ll  __MAX = 1'111'111'111'111'111'111LL; // prime
constexpr ll  __MIN = -__MAX;

//--------------------------------------------------------------------------------------------------

#include <vector>

int main() {
    FAST_IO;

    //   logic
    int n;
    cin >> n;
    vector<int> shelf(n);

    for (int i = 0; i < n; ++i) {
        cin >> shelf[i];
    }

    int cnt  = 0;
    int prev = 0;
    for (auto it = shelf.begin(); it != shelf.end(); ++it) {
        if (prev < *it) {
            cnt++;
        }

        if (prev < *it) {
            prev = *it;
        }
    }
    cout << cnt << "\n";

    cnt  = 0;
    prev = 0;

    for (auto it = shelf.rbegin(); it != shelf.rend(); ++it) {
        if (prev < *it) {
            cnt++;
        }
        if (prev < *it) {
            prev = *it;
        }
    }
    cout << cnt;

    return 0;
}
