#include <deque>
#include <iostream>
#include <string>

int main() {
    //   logic
    string sentence;
    string exp;

    cin >> sentence >> exp;

    deque<char> q;

    for (const char& c: sentence) {
        q.push_back(c);

        if (q.size() >= exp.size()) {
            string last_word(q.end() - exp.size(), q.end());

            if (last_word == exp) {
                q.erase(q.end() - exp.size(), q.end());
            }
        }
    }

    if (q.empty()) {
        cout << "FRULA";
    } else {
        for (const char& c: q) {
            cout << c;
        }
    }
}
