#include <vector>

void solution()
{
    int pay;
    cin >> pay;

    int change = 1000 - pay;

    vector<int> coin = {
        500,
        100,
        50,
        10,
        5,
        1
    };

    int count = 0;

    for (int i = 0; i < static_cast<int>(coin.size()); ++i) {
        count += change / coin[i];
        change %= coin[i];
    }

    cout << count;
}

void solution_old()

    int coin_pos = 0;

        if (change / coin[coin_pos] > 0) {

            change %= coin[coin_pos];

    }

}
