    int n, k;

    cin >> n >> k;
    vector<int> w(n);

    for (int i = 0; i < n; ++i) {
        cin >> w[i];
    }

    int cnt           = 0;
    int last_coin_pos = n - 1;

    while (k > 0 && last_coin_pos >= 0) {
        if ((k / w[last_coin_pos]) >= 1) {
            cnt += (k / w[last_coin_pos]);
            k %= w[last_coin_pos];
        }
        last_coin_pos--;
    }
    cout << cnt;
}
