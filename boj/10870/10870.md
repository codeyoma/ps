int fibo(int n) {
    if (n <= 2) {
        return 1;
    }
    return fibo(n - 1) + fibo(n - 2);
}

int dp(int n) {
    if (n <= 1) {
        return n;
    }

    int before = 1;
    int after  = 1;

    for (int i = 3; i <= n; ++i) {
        int temp = before;
        before   = after;
        after    = temp + before;
    }

    return after;
}
