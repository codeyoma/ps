    int                  n = 4;
    vector<vector<bool>> v(101, vector<bool>(101, false));

    while (n--) {
        int s_x, s_y, e_x, e_y;
        cin >> s_x >> s_y >> e_x >> e_y;

        for (int x = s_x; x < e_x; ++x) {
            for (int y = s_y; y < e_y; ++y) {
                v[x][y] = true;
            }
        }
    }

    int answer = 0;
    for (const auto& r: v) {
        for (const auto& c: r) {
            if (c) {
                answer++;
            }
        }
    }
    cout << answer;
}
