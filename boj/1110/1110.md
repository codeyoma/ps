    int count    = 0;

        int l_r, r_r;

        l_r = n % 10;
        r_r = (n / 10 + n % 10) % 10;

        n = l_r * 10 + r_r;
        count++;

        if (n == origin_n) {
            break;
        }
    }

    cout << count;
}
