#include <iostream>
#include <stack>
#include <string>

int main() {
    //   logic
    int n;
    cin >> n;

    stack<string> s;

    while (n--) {
        string cmd;
        cin >> cmd;

        if (cmd == "push") {
            cin >> cmd;
            s.push(cmd);
        } else if (cmd == "pop") {
            if (s.size() == 0) {
                cout << "-1\n";
            } else {
                cout << s.top() << "\n";
                s.pop();
            }
        } else if (cmd == "size") {
            cout << s.size() << "\n";
        } else if (cmd == "empty") {
            if (s.empty()) {
                cout << "1\n";
            } else {
                cout << "0\n";
            }
        } else if (cmd == "top") {
            if (s.size() == 0) {
                cout << "-1\n";
            } else {
                cout << s.top() << "\n";
            }
        }
    }
}
