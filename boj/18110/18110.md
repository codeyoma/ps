#include <cmath>
#include <vector>
void solution()
{
    int n;
    cin >> n;

    if (n == 0) {
        cout << 0;
        return;
    }
    vector<int> s(n);

    for (int i = 0; i < n; ++i) {
        cin >> s[i];
    }

    sort(s.begin(), s.end());

    int cut = static_cast<int>(round(n * 0.15));

    double sum = 0;
    for (int i = cut; i < n - cut; ++i) {
        sum += s[i];
    }

    cout << static_cast<int>(round(sum / (n - (cut * 2.0))));
}
