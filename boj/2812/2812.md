#include <iostream>
#include <stack>
#include <string>

int main() {
    //   logic
    int    n, k, count = 0;
    string number;
    cin >> n >> k >> number;

    stack<char> s;

    for (size_t i = 0; i < number.size(); ++i) {
        while (!s.empty() && s.top() < number[i] && count < k) {
            s.pop();
            count++;
        }
        s.push(number[i]);
    }

    while (count < k) {
        s.pop();
        count++;
    }

    stack<char> r_s;

    while (!s.empty()) {
        r_s.push(s.top());
        s.pop();
    }

    while (!r_s.empty()) {
        cout << r_s.top();
        r_s.pop();
    }
}
