using namespace std;
#define FAST_IO                  \
    ios::sync_with_stdio(false); \
    cin.tie(nullptr);

// typedef long long ll;
using ll = long long;

constexpr int _MAX  = 1'234'567'891; // prime
constexpr int _MIN  = -_MAX;
constexpr ll  __MAX = 1'111'111'111'111'111'111LL; // prime
constexpr ll  __MIN = -__MAX;

//--------------------------------------------------------------------------------------------------

#include <vector>

// multiset solution
ll  tele[300'001];
int bri[500'001];
int q, k;

void push(ll x) {
    if (k < x) {
        x = k;
    }

    if (x) {
        bri[x]++;
        q++;
    }
}

int main() {
    FAST_IO;

    ll  l;
    int n;
    cin >> l >> n >> k;

    for (int i = 0; i < n; ++i) {
        cin >> tele[i];
    }

    for (int i = 0; i < n && k; ++i) {
        cout << "0\n";
        k--;
    }

    push(tele[0]);
    push(l - tele[n - 1]);

    for (int i = 0; i < n - 1; ++i) {
        ll dist = tele[i + 1] - tele[i];

        push(dist / 2);
        push((dist - 1) / 2);
    }

    int i = 1;
    while (k) {
        for (int t = q; k && t;) {
            cout << i << "\n";
            k--;
            t--;
        }
        q -= bri[i++];
    }

    return 0;
}
