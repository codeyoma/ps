        for (int j = i + 1; j < n; ++j) {
            if ((sum - (s[i] + s[j])) == 100) {
                vector<int> answer;

                for (int k = 0; k < n; k++) {
                    if (k != i && k != j) {
                        answer.push_back(s[k]);
                    }
                }

                sort(answer.begin(), answer.end());

                for (const auto& c: answer) {
                    cout << c << "\n";
                }
            }
        }
    }
