#include <iostream>
#include <vector>

int main() {
    //   logic
    int a, b, answer = 0;
    cin >> a >> b;
    vector<bool> t(b + 1, true), a_t(b + 1, false);
    t[1] = false;

    for (int i = 2; i <= b; ++i) {
        if (t[i] == false) {
            continue;
        }

        for (int j = i + i; j <= b; j += i) {
            t[i] = false;
        }

        for (int j = i * i; j <= b; j *= i) {
            if (a_t[j] == false) {
                a_t[j] = true;
                answer++;
            }
        }
    }

    cout << answer;
}
