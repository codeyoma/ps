#include <vector>
void solution() {
    vector<int> timer = { 300, 60, 10 };
    vector<int> push_count(3, 0);

    int t;
    cin >> t;

    for (size_t i = 0; i < timer.size(); ++i) {
        push_count[i] = t / timer[i];
        t %= timer[i];
    }

    if (t) {
        cout << -1;
    } else {
        cout << push_count[0] << " " << push_count[1] << " " << push_count[2];
    }
}
