#include <vector>
void solution() {
    int n;
    cin >> n;
    vector<int> m(n, 0);

    for (int i = 0; i < n; ++i) {
        cin >> m[i];
    }

    int max_a = MIN;

    for (int i = 0; i < n; ++i) {
        int cnt = 0;
        for (int j = i + 1; j < n; ++j) {
            if (m[i] >= m[j]) {
                cnt++;
            } else {
                break;
            }
        }
        max_a = max(max_a, cnt);
    }
    cout << max_a;
}
