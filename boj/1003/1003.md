void fast_io() {

//--------------------------------------------------------------------------------------------------

#define MAX (1234567891)
#define MIN (-1234567891)

#include <iostream>
#include <vector>

struct base {
    int zero_count;
    int one_count;
};

int main() {
    fast_io();

    //   logic
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;

        vector<base> dp(n + 1);
        dp[0] = { 1, 0 };
        dp[1] = { 0, 1 };

        if (n >= 2) {
            dp[2] = { 1, 1 };

            for (int i = 3; i <= n; ++i) {
                dp[i] = { dp[i - 1].zero_count + dp[i - 2].zero_count,
                          dp[i - 1].one_count + dp[i - 2].one_count };
            }
        }

        cout << dp[n].zero_count << " " << dp[n].one_count << "\n";
    }
}
