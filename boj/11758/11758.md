typedef struct {
    int x;
    int y;
} coor;

int ccw(coor& a, coor& b, coor& c) {
    int r = (b.x - a.x) * (c.y - a.y) - (c.x - a.x) * (b.y - a.y);

    if (r == 0) {
        return 0;
    } else if (r > 0) {
        return 1;
    }
    return -1;
}

int main() {
    FAST_IO;

    vector<coor> c(3);

    for (int i = 0; i < 3; ++i) {
        cin >> c[i].x >> c[i].y;
    }
    cout << ccw(c[0], c[1], c[2]);

    END;
}
