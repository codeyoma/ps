#include <algorithm>
#include <vector>

void solution() {
    int n;
    cin >> n;

    vector<vector<int>> c(n, vector<int>(2, 0));

    for (int i = 0; i < n; ++i) {
        int a, t;

        cin >> a >> t;
        c[i][0] = a;
        c[i][1] = t;
    }

    sort(c.begin(), c.end(), [](const vector<int>& a, const vector<int>& b) {
        return a[0] < b[0];
    });

    for (const auto& i: c) {
        for (const auto& j: i) {
            _log << j << " ";
        }
        _log << endl;
    }

    int time = 0;

    for (const auto& i: c) {
        if (time < i[0]) {
            time += i[0] - time;
        }
        time += i[1];
    }
    cout << time;
}
