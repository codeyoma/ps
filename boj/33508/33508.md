answer = 0

left_table = set()
right_table = [0] * (n + 1)

for x in a:
	right_table[x] += 1

for x in a:

	if right_table[x] == 2:
		if x in left_table:
			answer += len(left_table) - 1
		else:

	right_table[x] -= 1
